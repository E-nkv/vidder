// builds.txt

# Build for Linux 64-bit
GOOS=linux GOARCH=amd64 go build -o ./bin/vidder-linux-amd64 .

# Build for Linux 32-bit
GOOS=linux GOARCH=386 go build -o ./bin/vidder-linux-386 .

# Build for Windows 64-bit
GOOS=windows GOARCH=amd64 go build -o ./bin/vidder-windows-amd64.exe .

# Build for Windows 32-bit
GOOS=windows GOARCH=386 go build -o ./bin/vidder-windows-386.exe .

# Build for macOS 64-bit Intel
GOOS=darwin GOARCH=amd64 go build -o ./bin/vidder-darwin-amd64 .

# Build for macOS ARM64 (Apple Silicon)
GOOS=darwin GOARCH=arm64 go build -o ./bin/vidder-darwin-arm64 .
